Ecrire le programme qui calcule la plus grande et la plus petite suite croissante de valeurs d'une série de nombres entrés au clavier.                                         // petiteSuite   //grandeSuite

- La fin de la série sera annoncée par un 100.  // stop
- Le programme affichera également :
  - la position du premier maximum;     // firstPosMax
  - la position du dernier maximum;     // lastPosMax
  - le nombre d'occurrence du maximum;  // occurMax
  - la position du premier minimum;     // firstPosMin
  - la position du dernier minimum;     // lastPosMax
  - le nombre d'occurrence du minimum;

************* VERSION FULL OPTION *************

Debut EX44
    VARIABLE LOCALES :
        VAR input :N <- 0
        VAR min :N <- 0
        VAR max :N <- 0
        VAR firstPosMin :N <- 1
        VAR lastPosMin :N <- 1
        VAR firstPosMax <- 1
        VAR lastPosMax <- 1
        VAR occurMin <- 1
        VAR occurMax <- 1
        VAR suiteMin <- 0 // on pars du principe qu'une suite est constitué d'au moins 2 nombres croissants consecutifs
        VAR suiteMax <- 0
        VAR suite <- 0
        VAR previous <- 0
        VAR i :N <- 0
        CONST stop :N <- 100
    FIN VARIABLES LOCALES

    Afficher "Saisir une serie de nombre"
    Afficher "Saisir ", stop, " pour terminer"
    Saisir input
    min <- input
    max <- input

    TANT QUE input ≠ stop
        i <- i + 1
        SI i ≠ 1
        ALORS
            SI input ≤ min
                SI input = min
                ALORS
                    occurMin <- occurMin + 1
                    lastPosMin <- i
                SINON
                    min <- input
                    occurMin <- 1
                    firstPosMin <- i
                    lastPosMin <- i
                FINSI
            FINSI
            SI input ≥ max
                SI input = max
                ALORS
                    occurMax <- occurMax + 1
                    lastPosMax < i
                SINON
                    max <- input
                    occurMin <- 1
                    firstPosMax <- i
                    lastPosMax <- i
                FINSI
            FINSI

//calcul des suites croissantes

            SI input > previous                             //ici on incremente suite qui est en cours
            ALORS
                SI suite = 0
                ALORS
                    suite <- 2                              // 2 car on detecte une suite a partir de 2 nombres croissants successifs
                SINON
                    suite <- suite + 1
                FINSI
            FINSI
            SI input < previous                             // on met a jour suiteMin et suiteMax que si
                SI suite ≠ 0                                                                           //une suite en cours a ete rompue
                    SI suiteMin = 0
                    ALORS
                        suiteMin <- suite
                    SINON
                        SI suiteMax = 0
                            SI suite < suiteMin
                            ALORS
                                suiteMax <- suiteMin
                                suiteMin <- suite
                                SINON
                                    suiteMax <- suite
                            SINON
                                SI suite < suiteMin
                                ALORS
                                    suiteMin <- suite
                        FINSI
                        SI suite > suiteMax
                        ALORS
                            suiteMax <- suite
                        FINSI
                    FINSI
                    suite <- 0
                FINSI
            FINSI
        FINSI
        previous <- input

        Afficher "Saisir une serie de nombre"
        Afficher "Saisir ", stop, " pour terminer"
        Saisir input
    FIN TANT QUE

    SI suite ≠ 0                                             recopie //je mets a jour suiteMin et suiteMax si une suite etait en cours juste avant de terminer (stop = 100)
        SI suiteMin = 0
        ALORS
            suiteMin <- suite
        SINON
            SI suiteMax = 0
                SI suite < suiteMin
                ALORS
                    suiteMax <- suiteMin
                    suiteMin <- suite
                    SINON
                        suiteMax <- suite
                SINON
                    SI suite < suiteMin
                    ALORS
                        suiteMin <- suite
            FINSI
            SI suite > suiteMax
            ALORS
                suiteMax <- suite
            FINSI
        FINSI
    FINSI


    CAS OU i
        CAS 0
            Afficher "Vous n'avez saisi aucun nombre"
        CAS 1
            Afficher "Vous n'avez saisi qu'un seul nombre qui est :", input
        AUTRES CAS
            Afficher "Le plus petit nombre est ", min, ", et a ete saisi ", occurMin, "fois dans cette serie .."
            SI occurMin = 1
            ALORS
                Afficher "..en position :", firstPosMin
            SINON
                Afficher "..le 1er en position ", firstPosMin, " et le dernier detecté en position ", lastPosMin
            FINSI
            Afficher "Le plus grand nombre est ", max, ", et a ete saisi ", occurMax, "fois dans cette serie .."
            SI occurMax = 1
            ALORS
                Afficher "..en position :", firstPosMax
            SINON
                Afficher "..le 1er en position ", firstPosMax, " et le dernier detecté en position ", lastPosMax
            FINSI

// ajout info suite

            SI suiteMin 0 ET suiteMax = 0
            ALORS
                Afficher "Il n'y a pas de suite croissante dans la serie de nombres entres"
            FINSI
            SI suiteMin ≠ 0 ET suiteMin = suiteMax OU SuiteMax ≠ 0 ET suiteMin = suiteMax OU suiteMin ≠ 0 ET suiteMax = 0
            ALORS
                Afficher "Il n'y a qu'une suite de nombres croissants, elle compte :", suiteMin, "nombres croissants consecutifs"
            FINSI
            SI suiteMin ≠ 0 ET suiteMax ≠ 0 ET suiteMin < suiteMax
            ALORS
                Afficher "La plus petite suite compte", suiteMin, "nombres, et la plus grande compte ", suiteMax, "nombres."
    FIN CAS OU

  Fin EX44
