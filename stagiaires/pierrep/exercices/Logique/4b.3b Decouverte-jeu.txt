Debut Decouverte
    VAR LOCALES

    x :N <- 0
    r :N <- 0
    max :N <- 0

    FIN VAR LOCALES

    affiche "Entrez un seuil en deça duquel un nombre sera généré aléatoirement"

    saisie max 

    r <- fonction aléatoire(max : N):N

    affiche "Devinez un nombre compris entre 0 et", max

    saisir x

    TANT QUE x != r

        SI x < r

            affiche "le nombre à trouver est supérieur au nombre que vous avez entré"

        ELSI x > r
    
            affiche "le nombre à trouver est inférieur au nombre que vous avez entré"
    
    saisir x
    
    FIN TANT QUE

    affiche "Bravo"
 
Fin Decouverte
