PROGRAMME

    VARIABLE LOCALE	

    var entry-user-text :T <- ""
    var entry-user-char :C <- ""
    var first-char-index <- 0
    var last-char-index <- 0
    var occurences :N <- 0
    var compteur :N <- 0
    var longueur-texte :N <- longeur(entry-user-text :T):N

    FIN VARIABLE LOCALE 

    AFFICHER("Entrer texte")
    SAISIR entry-user-text

    AFFICHER("Entrer charactere")
    SAISIR entry-user-char

    POUR compteur ALLANT DE 1 A longueur-texte

        // PREMIER INDEX - SI PAS D'OCCURENCES => LE 1ER INDEX DEVIENT LE NUMERO DE COMPTEUR

        SI(caract(entry-user-text, compteur) = entry-user-char ET occurences = 0)
         ALORS 
            first-char-index <- compteur
            occurences <- occurences + 1
        FIN SI

        // DERNIER INDEX
        // [NOTE] LE PREMIER INDEX EST DEJA DEFINI AVEC LE PREMIERE CONDITION - POUR RENTRER DANS CELLE-CI IL FAUT QUE OCCURENCES SOIT SUPERIEUR OU EGAL A 1
        // [DERNIER INDEX] SI L'INDEX SUIVANT EST POSITIF A LA VERIFICATION - CELUI-CI DEVIENT LE NOUVEAU last-char-index

        SI (caract(entry-user-text, compteur) = entry-user-char ET occurences >= 1)
         ALORS 
                last-char-index <- compteur
                occurences <- occurences + 1
        FIN SI

    FIN POUR

    AFFICHER("Le caractere" + entry-user-char + "est en premiere position a l'index", first-char-index)
    AFFICHER("Le caractere" + entry-user-char + "est en derniere position a l'index", last-char-index)
    AFFICHER("Le caractere apparait : " + occurences + "fois en tout.")

FIN PROGRAMME
