Ecrire le programme qui déplace les valeurs d’un tableau carré d’un quart de tour dans le sens horlogique.
• Version 1 : utilisation d’un tableau intermédiaire de même taille
• Version 2 : utilisation d’une seule variable intermédiaire. 

DEBUT 7-5a
    VARIABLES LOCALES :
        CONST n:N <- 6
        VAR tab :N[n, n]
        VAR tab2 :N[n, n]
        VAR i :N <- 1
        VAR j :N <- 1
    FIN VARIABLES LOCALES

    i <- 1
    POUR i ALLANT DE 1 A n
        j<- 1
        POUR j ALLANT DE 1 A n
            tab2[j, n+1-i] <- tab[i,j] 
        j SUIVANT
    i SUIVANT

FIN 7-5a


DEBUT 7-5b
    VARIABLES LOCALES :
        CONST n:N <- 6
        VAR tab :N[n, n]
        VAR int :N <- 0
        VAR k :N <- 0
        VAR i :N <- 1
        VAR j :N <- 1
        VAR l :N <- 1
        VAR cpt :N 
    FIN VARIABLES LOCALES

    cpt <- 1
    POUR cpt ALLANT DE 1 à n-1
        i <- 1
        j <- cpt
        int <- tab[i,j]
        TANTQUE (k < 4)
            tab[i-j] <- tab[n+1-j, i]
            l <- i
            i <- n+1-j
            j <- l
            k <- k + 1
        FIN TANT QUE
        tab[i,j] <- int
    cpt SUIVANT
    
FIN 7-5b
