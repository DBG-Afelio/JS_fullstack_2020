    VARIABLES LOCALES :
        var nombre :N <- 0
        var factorielle :N 
        var i: N
    FIN VARIABLES LOCALES

    Afficher "Entrez un nombre"
    Saisir nombre

    
    factorielle <- 0
    i <- nombre
    TANT QUE (i > 0)
        factorielle <- factorielle * i
        i <- i - 1
    FIN TANT QUE

    Afficher "La factorielle de ", nombre, " vaut ", factorielle
