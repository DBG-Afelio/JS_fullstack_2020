// Ecrire le programme  qui initialise les cases de la diagonale principale 
// d’un tableau carré avec la valeur 1 et les cases de la diagonale secondaire 
// avec la valeur 2.


Début diagonaleCarre

    Variables locales
        var carre: N[A, A]
        reste: N <-0
        centre: N <-0

    Fin variables locales

    Si (longueur(A)%2) = 0
        Alors
            Pour i allant de 1 à longueur(A)
                carre[i, i] <- 1
            i suivant
        
            Pour j allant de 1 à longueur(A)
                carre[i, (longueur(A)+1)-i] <- 2
            j suivant

    Sinon

            Pour i allant de 1 à longueur(A)
                carre[i, i] <- 1
            i suivant
        
            Pour j allant de 1 à longueur(A)
                carre[i, (longueur(A)+1)-i] <- 2
            j suivant

            reste = longueur(A)DIV2
            centre = reste +0.5

            carre[centre, centre] <- 3

    Fin Si

    Afficher carre

Fin diagonaleCarre
