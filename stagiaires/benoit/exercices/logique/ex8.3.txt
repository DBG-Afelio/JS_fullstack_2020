// Ecrire le programme qui affiche les nombres parfaits inférieurs à 100.
// Un nombre parfait est un entier égal à la somme de ses diviseurs sauf lui-même.
// 6  -> 1, 2, 3 et 6, et il vérifie bien 2 × 6 = 12 = 1 + 2 + 3 + 6


Début nombresParfait

    Variables locales

        var debut: N -> 0
        var fin: N -> 100
        var numero: N -> 0
        var compteur: N -> 1
        var taille: N -> 1
        var tab: N[taille]

    Fin variables locales




    Pour i allant de debut à fin

        Pour j allant de 1 à j
            
            i/j = numero
            
            Si numero MOD 1 = 0
                tab[compteur] = numero
                taille++
                compteur++
            Fin Si    
        
        j suivant

        Pour k allant de 1 à longueur(tab)
            somme = somme + tab[k]
        k suivant

        Si (2xi = somme)
            Alors 
                Afficher i
        FinSi
        
    i suivant   

Fin nombresParfait
